### ⭐测试次数

19

**DP 1**

`dp[i][j]` 表示剩余手机数 i 和楼层高度 j 所需的测试次数。

当前选择第 `k` 层楼摔下手机，则结果可能是摔坏或者没摔坏，且测试次数加一，则状态转移方程如下：

`dp[i][j] = min( dp[i][j], max(dp[i][j-k], dp[i-1][k-1])+1 )`

```cpp
#include<bits/stdc++.h>
using namespace std;

int dp[4][1005];

int main(){
  memset(dp, 0x3f, sizeof(dp));
  // 仅有一部手机，最佳策略是从楼底一层层往上摔
  for(int i=1; i<=1000; i++) dp[1][i]=i;
  for(int i=2; i<=3; i++){
    // 状态转移时会使用
    dp[i][0]=0;
    for(int j=1; j<=1000; j++){
      for(int k=1; k<=j; ++k){
        dp[i][j]=min(dp[i][j], max(dp[i][j-k], dp[i-1][k-1])+1);
      }
    }
  }
  cout<<dp[3][1000];
  return 0;
}
```

**DP2**

`dp[i][j]` 表示测试次数 i 和剩余手机数 j 最多能测试的楼层高度。

若当前楼层 `k` 摔下手机，结果如下：

- 摔坏了，则 `curHeight = maxHeight = dp[i-1][j-1] = k - 1`
- 没摔坏，则 `curHeight = dp[i-1][j] + 1`

状态转移方程如下：

`dp[i][j] = dp[i-1][j-1] + dp[i-1][j] + 1`

### ⭐日志统计

**滑动窗口**

将数据按照先 id 后时间的顺序升序排序，双指针维护一个满足“热帖”要求的时间区间，统计单个 id 在该区间内的点赞数即可。

```cpp
#include<bits/stdc++.h>
using namespace std;

typedef pair<int, int> PII;
#define id first
#define ts second

PII a[100005];
int ans[100005];

int main(){
  int n,d,k;
  cin>>n>>d>>k;
  for(int i=0; i<n; i++){
    cin>>a[i].ts>>a[i].id;
  }
  sort(a, a+n);
  memset(ans, 0, sizeof(ans));
  // i 在前面，j 在后面
  int i=0, j=0;
  while(j<n){
    // 对单个 id 统计
    int t=a[i].id;
    int cnt=0;
    while(a[j].id==t && j<n){
      // 满足时间间隔，点赞数增加
      if(a[j].ts-a[i].ts<d) ++cnt,++j;
      // 时间间隔太久，i 后移尝试近一点的时间
      else --cnt,++i;
      // 找到热帖
      if(cnt>=k){
        // 将指针移动到下一个 id
        while(a[j].id==t && j<n) ++j;
        // 更新 ans[id] 状态
        ans[t]=1;
      }
    }
    i=j;
  }
  for(int i=0; i<=100000; i++){
    if(ans[i]){
      printf("%d\n", i);
    }
  }
  return 0;
}
```

```cpp
char mp[1005][1005];
bool vis[1005][1005]; // 仅标记访问过的陆地
int D[] = {-1,0,1,0,0,-1,0,1};
int n, ans, hasLand;

int dfs(int x, int y){
  // 该区域是水域
  if(mp[x][y]=='.') return 0;
  // 该区域是陆地
  vis[x][y]=1;

    if(nx>=0&&nx<n && ny>=0&&ny<n && !vis[nx][ny]){
      // 四周有水域就会被淹没
      flag&=dfs(nx, ny);
    }
  }
  if(flag) hasLand=1;
  return 1;
}

int main(){
  cin>>n;
  for(int i=0; i<n; i++)
    cin>>mp[i];
  memset(vis, 0, sizeof(vis));
  ans=0;
  for(int i=1; i<n; i++){
    for(int j=1; j<n; j++){
      if(!vis[i][j] && mp[i][j]=='#'){
        hasLand=0;
        dfs(i,j);
        if(!hasLand) ++ans;
      }
    }
  }
  cout<<ans;
  return 0;
```

```cpp
int D[] = {-1,0,1,0,0,-1,0,1};
int n;

int bfs(int x, int y){
  queue<PII> q;
  q.push({x,y});
  int allCnt=0, realCnt=0;
  while(q.size()){
    ++allCnt;
    PII cur=q.front(); q.pop();
    x=cur.first, y=cur.second;
    vis[x][y]=1;
    int flag=0;
    for(int i=0; i<4; i++){
      int nx=x+D[i<<1], ny=y+D[i<<1|1];
      if(nx>=0&&nx<n&&ny>=0&&ny<n){
        if(mp[nx][ny]=='.') flag=1;
        else if(!vis[nx][ny]) q.push({nx,ny});
      }
    }
    if(flag) ++realCnt;
  }
  return realCnt==allCnt;
}

int main(){
  cin>>n;
  for(int i=0; i<n; i++)
    cin>>mp[i];
  memset(vis, 0, sizeof(vis));
  int ans=0;
  for(int i=1; i<n; i++){
    for(int j=1; j<n; j++){
      if(!vis[i][j] && mp[i][j]=='#'){
        ans += bfs(i,j);
      }
    }
  }
  cout<<ans;
  return 0;
}
```

### ✔️螺旋折线

将螺旋的折线看成是多个中心在原点处的正方形，判断落点的位置，统计外圈和内部正方形的周长总和即可。

```cpp
#include<bits/stdc++.h>
using namespace std;

typedef unsigned long long ULL;

int main(){
  int x,y,ux,uy;
  cin>>x>>y;
  ux=abs(x), uy=abs(y);
  // 落点的正方形 1/2 边长
  int t = max(ux,uy);
  if(!t) {cout<<0; return 0;}
  // 计算内圈长度
  ULL ans = 4ull*t*(t-1);
  // 确定落点的边界，计算外圈长度
  // 落在左下角的交点应按照下边界处理，所以处理顺序应该先下后左。
  // 上边界
  if(y==t) ans += 2ull*t+x+t;
  // 下边界
  else if(y==-t) ans += 6ull*t+t-x;
  // 左边界
  else if(x==-t) ans += t+y;
  // 右边界
  else ans += 4ull*t+t-y;
  cout<<ans;
  return 0;
}
```

```cpp
  cntB[0]=0;  // 求前缀和要删除 b 取 0 的情况，因为这时候找不到更小的 a
  for(int i=1; i<=100000; i++){

  for(int i=0; i<n; i++){
```

### ✔️等腰三角形

三角形边长与高度关系 `C = 4*(H-1)`。

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
  int h, c;
  scanf("%d", &h);
  c=4*(h-1);
  string S, T;
  for(int i=1,j=0; j<c; ++i){
    int t=i;
    T.clear();
    while(t){
      T.push_back(t%10+'0');
      t/=10;
      ++j;
    }
    reverse(T.begin(),T.end());
    S+=T;
  }
  int p1=0, p2=c-1;
  for(int i=0; i<h; i++){
    for(int j=h-i-1; j>0; j--) putchar('.');
    int l=h-i, r=h+i;
    if(l==r) putchar(S[p1++]);
    else if(i+1==h){
      for(int j=p1; j<=p2; j++) putchar(S[j]);
    }else{
      putchar(S[p1++]);
      for(int j=l+1; j<r; j++) putchar('.');
      putchar(S[p2--]);
    }
    putchar('\n');
  }
}
```
