## 编程

### ✔️刷题统计

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
  ll a, b, n;
  cin >> a >> b >> n;
  ll week = a * 5 + b * 2;
  ll ans = n / week * 7;
  ll d = -1, cnt = n / week * week;
  while (cnt < n) {
    d = (d + 1) % 7;
    ++ans;
    if (d == 5 || d == 6) cnt += b;
    else cnt += a;
  }
  cout << ans;
  return 0;
}
```

### ❌修剪灌木

**规律**

特判 `n = 1` 的情况。

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
  int n;
  cin >> n;
  if (n == 1) {
    cout << 1;
    return 0;
  }
  for (int i = 1; i <= n; i++) {
    cout << max(2ll * (i - 1), 2ll * (n - i)) << '\n';
  }
  return 0;
}
```

### ⭐X 进制减法

考查对进制的理解。

```
num  16  8 10  2
val   7  3  2  1

DEC: 7 * (8*10*2) + 3 * (10*2) + 2 * (2) + 1
```

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int a[100005], b[100005];
int la, lb;
ll base[100005];
const int mod = 1000000007;

int main()
{
  ios::sync_with_stdio(0);
  int n;
  cin >> n;
  cin >> la;
  for (int i = la - 1; i >= 0; --i)
    cin >> a[i];
  cin >> lb;
  for (int i = lb - 1; i >= 0; --i)
    cin >> b[i];
  base[0] = 1;
  for (int i = 0; i < la; i++) {
    base[i + 1] = base[i] * min(max(max(a[i], b[i]) + 1, 2), n) % mod;
  }
  ll A = 0, B = 0;
  for (int i = 0; i < la; i++) {
    A += base[i] * a[i] % mod;
    A %= mod;
  }
  for (int i = 0; i < lb; i++) {
    B += base[i] * b[i] % mod;
    B %= mod;
  }
  cout << (A + mod - B) % mod;
  return 0;
}
```

### ⭐统计子矩阵

**暴力（部分样例）**

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int a[505][505], sum[505][505];

int main()
{
  ios::sync_with_stdio(0);
  int n, m, k;
  cin >> n >> m >> k;
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
      cin >> a[i][j];
      sum[i][j] = a[i][j] + sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1];
    }
  }
  ll ans = 0;
  for (int lx = 1; lx <= n; lx++)
    for (int ly = 1; ly <= m; ly++)
      for (int rx = lx; rx <= n; rx++)
        for (int ry = ly; ry <= m; ry++)
          if (sum[rx][ry] - sum[lx - 1][ry] - sum[rx][ly - 1] + sum[lx - 1][ly - 1] <= k)
            ++ans;
  cout << ans;
  return 0;
}
```
