```cpp
#include <bits/stdc++.h>
using namespace std;

int a[205];     // 1 已选，0 未选
int sq[205];    // 标记完全平方数
int ans;

// k 指当前选择的元素，r1 指当前已选择元素个数，r2 指当前还能选择的元素个数
void dfs(int k, int r1, int r2){
  if(r1+r2<=ans) return;
  if(k>100 || r2==0){
    ans=max(ans,r1);
    return;
  }
  if(a[k]){
    // 选择元素 k，则 k 之后构成完全平方的数不能选
    int v[100], sz=0;
    for(int i=k+1; i<=100 && i+k<=199; i++){
      if(sq[i+k] && a[i]){
        a[i]=0;
        v[sz++]=i;
      }
    }
    dfs(k+1,r1+1,r2-sz-1);
    for(int i=0; i<sz; i++) a[v[i]]=1;
    a[k]=0;
    dfs(k+1,r1,r2-1);
    a[k]=1;
  }else{
    dfs(k+1,r1,r2);
  }
}

int main(){
  memset(sq,0,sizeof(sq));
  for(int i=1; i*i<=199; i++) sq[i*i]=1;
  for(int i=1; i<=100; i++) a[i]=1^sq[i];
  int rest=0;
  for(int i=1; i<=100; i++){
    if(a[i]) ++rest;
  }
  ans=0;
  dfs(1,0,rest);
  cout<<ans;
  return 0;
}
```

### ⭐✔️拼接

**DFS**

一开始思维限制在搜索小方块的拼接路径，情况繁琐。改为搜索切割的路径，从左侧或上侧对称切割到对角线即算成功，思路豁然开朗。

我们将大正方形放入笛卡尔坐标系，左下角记为 `(0, 0)`，右上角记为 `(7, 7)` 求解。

```cpp
#include <bits/stdc++.h>
using namespace std;

int vis[8][8];
int dx[]={-1,1,0,0};
int dy[]={0,0,-1,1};
int ans;

void dfs(int x, int y){
  // 切到对角线，分割成功
  if(x==y){
    ++ans;
    return;
  }
  vis[x][y]=1;
  for(int i=0; i<4; i++){
    int nx=x+dx[i], ny=y+dy[i];
    if(nx<0 || nx>7 || ny<0 || ny>7 || vis[nx][ny]) continue;
    if(nx==0 || ny==7) continue;  // 切到边缘，分割失败
    dfs(nx,ny);
  }
  vis[x][y]=0;
}

int main(){
  ans=0;
  memset(vis,0,sizeof(vis));
  for(int i=1; i<7; i++) dfs(0,i);  // 横着开始切
  for(int i=1; i<7; i++) dfs(i,7);  // 竖着开始切
  cout<<ans+2;  // 加上整个大正方形属于同一边的情况
  return 0;
}
```

### ⭐✔️质数拆分

**DP**

对于每个质数 i，它与之前的数字 j 会产生新的组合 `i + j = t`，若用 `dp[x]` 表示 x 的质数拆分，则该组合记为：

```
dp[t] += dp[j]
```

此时还需要注意，若 j 本身也是一个质数，那么 i 除了与 j 的质数拆分进行组合，还可以和 j 本身进行组合。

```
dp[t] ++
```

注意状态压缩的代码实现与上述描述略有不同。

```cpp
#include <bits/stdc++.h>
using namespace std;

int p[2020];
long long dp[2020];

int main(){
  memset(p,0,sizeof(p));
  p[0]=p[1]=1;
  memset(dp,0,sizeof(dp));
  for(int i=2; i<2020; i++){
    if(!p[i]){
      for(int j=2019; j>i; j--){  // 反向更新 dp 状态
        int t=j-i;
        dp[j]+=dp[t];
        if(t<i && !p[t]) ++dp[j];
      }
      for(int j=2*i; j<2020; j+=i){
        p[j]=1;
      }
    }
  }
  cout<<dp[2019];
  return 0;
}
```

### ✔️最优旅行

41613

**DFS**

```cpp
#include <bits/stdc++.h>
using namespace std;

unordered_map<string,int> mp;
vector<string> cities = {"北京","上海","广州","长沙","西安","杭州","济南","成都","南京","昆明","郑州","天津","太原","武汉","重庆","南昌","长春","沈阳","贵阳","福州"};
const int sz = cities.size();
const int DAY = 24*60;

struct Edge {
  // l, r 时间单位是 min
  int to, l, r;
};

vector<Edge> G[30]; // 邻接表
int vis[30];
int ans;

void read(){
  string ss,su,sv;
  int l, r, hh, mm;
  while(cin>>ss>>su>>sv){
    scanf("%d:%d",&hh,&mm);
    l=60*hh+mm;
    scanf("%d:%d",&hh,&mm);
    r=60*hh+mm;
    // 数据没有 r < l 的情况
    G[mp[su]].push_back({mp[sv],l,r});
  }
}

// 到达城市 u，旅游城市数量 cnt，当前时间 cur，总时间 tot
void dfs(int u, int cnt, int cur, int tot){
  if(tot>ans) return;
  if(cnt==sz){
    ans=min(ans,tot);
    return;
  }
  for(auto v: G[u]){
    if(v.to==0 && cnt+1==sz || v.to && !vis[v.to]){
      vis[v.to]=1;
      int tmp=tot;
      // tot+=DAY;
      if(cur>v.l) tmp+=DAY; // 错过动车要再等一天
      tmp+=v.r-cur;
      dfs(v.to, cnt+1, v.r, tmp);
      vis[v.to]=0;
    }
  }
}

int main(){
  freopen("in.txt","r",stdin);
  for(int i=0; i<sz; i++) mp[cities[i]]=i;
  read();
  ans=1e9;
  memset(vis,0,sizeof(vis));
  dfs(0, 0, DAY/2, 0);
  // 加上城市的停留时间，原本停留时间是在 dfs 内累加的
  // 但这样做会增大 tot，使 tot < ans 的剪枝优化作用不明显
  cout<<ans+DAY*(sz-1);
  return 0;
}
```

### ⭐序列求和

292809912969717649

**数论**

唯一分解定理，若一个数质因数分解为 `p1^a1 * p2^a2 * ... * pn^an`，那么它因子个数可以通过指数得到。

```
(a1+1) * (a2+1) * ... * (an+1)
```

该题将约数个数 1 到 60 进行因式分解 `x1*x2*...*xn`，则分解（注意不是质因数分解，而是因数分解）对应的数可以表示为：

```
2^(x1-1) * 3^(x2-1) * 5^(x3-1) * ...
```

其中 `x1` 到 `xn` 应按照降序排列，以得到这种分解方式中最小的数。

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int N=1000;
int p[N+5];
vector<int> v;  // 质数序列
int fac[6];     // 因式分解的每个因子
int fi;         // 标记 fac[] 的长度
ll mn;          // 每次因式分解中最小的数

// 素数筛
void prime(){
  memset(p,0,sizeof(p));
  p[0]=p[1]=1;
  for(int i=2; i<=N; i++){
    if(!p[i]){
      v.push_back(i);
      for(int j=2*i; j<=N; j+=i){
        p[j]=1;
      }
    }
  }
}

// 快速幂
ll ksm(ll a, ll b){
  ll r=1;
  while(b){
    if(b&1) r*=a;
    a*=a, b>>=1;
  }
  return r;
}

// 计算该种因式分解的值，并更新 mn
void calc(int x){
  int a[7];
  for(int i=0; i<fi; i++) a[i]=fac[i];
  a[fi]=x;
  sort(a,a+fi+1,greater<int>());
  ll tmp=1;
  for(int i=0; i<=fi; i++){
    tmp*=ksm(v[i],a[i]-1);
  }
  mn=min(mn,tmp);
}

// 对 x 进行因式分解
void dfs(int x){
  if(x==1 || !p[x]){
    calc(x);
    return;
  }
  for(int i=2; i*i<=x; i++){
    if(x%i) continue;
    int t=x/i;
    fac[fi]=i, ++fi;
    calc(t);  // 统计不分解 t 的答案
    dfs(t);   // 统计继续分解 t 的答案
    --fi;
  }
}

int main(){
  prime();
  ll ans=0;
  for(int i=1; i<=60; i++){
    fi=0, mn=~(1ll<<63);
    dfs(i);
    ans+=mn;
  }
  printf("%lld", ans);
  return 0;
}
```

**数论**

扩展 GCD，假设 `t = (p - 1) * (q - 1)`，则 `(d * e) % t = 1` 可以转化为不定方程 `d * e - t * y = 1` 解出私钥。

扩展欧几里得在已知 `ax + by = gcd(a, b)` 下求解 x, y 的解：

1. 由 `gcd(a, b) = gcd(b, a % b)` 得 `ax + by = bx + (a % b)y`，因为递归过程中 a, b 不能改变，因此每次更新的是 x, y。
2. 递归终止条件为 `b = 0`，此时 `ax = gcd(a, b)`。

```
  bx + (a % b)y
= bx + (a-(a/b)*b)y
= bx + ay-(a/b)*b*y
= ay + b(x-a/b*y)
```

通过上述化简，求出递推式 `exgcd(ax + by) = exgcd(ay + b(x-a/b*y))`。

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll n,d,e,p,q,C,X;

void find_P_and_Q(){
  for(ll i=3; i*i<=n; i+=2){
    if(n%i==0){
      p=i, q=n/i;
      break;
    }
  }
}

ll exgcd(ll a, ll b, ll &x, ll &y){
  if(!b){
    x=1, y=0;
    return a;
  }
  ll r=exgcd(b,a%b,x,y), t=x;
  x=y, y=t-a/b*x;
  return r;
}

// 由于 ksm 中的乘法会超出 long long 范围
ll ksc(ll a, ll b, ll p){
  ll r=0;
  while(b){
    if(b&1) r=(r+a)%p;
    a=(a<<1)%p, b>>=1;
  }
  return r;
}

ll ksm(ll a, ll b, ll p){
  ll r=1;
  a%=p;
  while(b){
    if(b&1) r=ksc(r,a,p);
    a=ksc(a,a,p), b>>=1;
  }
  return r;
}

int main(){

  }
```
```cpp
}
```

### ❌数正方形

对于水平正方形，规律比较好找。

```
a(n) = n^2 + (n-1)^2 + (n-2)^2 + ... + 1^2
```

对于斜正方形讨论如下：

假设方格边长为 N，那么可以构造出新的 N-1 个内接正方形（四条边依次选取 1 到 N-1 作为正方形的四个顶点）。

假设 `f(n)` 表示边长为 N 的内接正方形个数，`b(n)` 表示边长为 N 的斜正方形个数。

```
b(2) = f(2) = 1
b(3) = f(2)*(3-1)^2 + f(3)
b(4) = f(2)*(4-1)^2 + f(3)*(4-2)^2 + f(4)
...
b(n) = 1*(n-1)^2 + 2*(n-2)^2 + ... + (n-1)*1^2
```

通过以上两个递推式即可计算出答案。

```cpp
#include <bits/stdc++.h>
using namespace std;

const int D=1e9+7;

int main(){
  int n;
  cin>>n;
  --n;
  // 水平正方形
  int a=0;
  for(int i=1; i<=n; i++){
    a=(a+(1ll*i*i)%D)%D;
  }
  // 斜正方形
  int b=0;
  for(int i=1; i<n; i++){
    b=(b+(1ll*i*(n-i)*(n-i))%D)%D;
  }
  cout<<(a+b)%D;
  return 0;
}
```
