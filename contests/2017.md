int s[20], c[20];
int pos[20], vis[20];
int x,y,z;

void dfs(int k, int v){
  pos[k]=v;
  vis[v]=1;
  int rx=0,ry=0,rz=0;
  for(int i=1; i<=k; ++i){
    int t=pos[i];
    rx+=t, ry+=s[t], rz+=c[t];
  }
  if(rx==x&&ry==y&&rz==z){
    int flag=vis[1];
    for(int i=1; i<=16; i++){

    }
    exit(0);

    vis[v]=0;
    return;
  }
  // 组合问题，默认有序避免重复计算
  for(int i=v+1; i<=16; i++){
    if(!vis[i]) dfs(k+1, i);
  }
  vis[v]=0;
}

  }

  x>>=1, y>>=1, z>>=1;

int n, m, ans;

    }

  g[cnt][x]=0;

      ++pos, ans=max(ans, dfs());
      // 返回最大值
      return ans;
    }else{
      // 累加
      ++ans, ++pos;
    }
  }
  return ans;
}

int main(){
  scanf("%s", s);
  pos=0;
  printf("%d", dfs());
  return 0;
}

  int y,m,d;
  bool operator< (const Date &x) const{

    if(y==x.y) return m<x.m;
    return y<x.y;
  }

{
  int _y,y,m,d;

  swap(m,d);
  if(fmt(y,m,d)) st.insert({y,m,d});

### ⭐包子凑数

**数论**

裴蜀定理，降序序列满足 `gcd(a0,a1,...,an)=1`，那么它们不能凑出的最大值一定小于等于 `(a1-1)*(a0-1)-1`。

将问题转化为完全背包，每种蒸笼对应每种物品，物品数量无限，要凑的数相当于背包的容量。背包问题求解尽可能装满，所以背包容量正好装满，就说明可以凑出该数。

```cpp
#include<bits/stdc++.h>
using namespace std;

int w[105], dp[10005];

int gcd(int a, int b){
  return b ? gcd(b,a%b) : a;
}

int main(){
  int n;
  scanf("%d", &n);
  int d=0;
  for(int i=0; i<n; i++){
    scanf("%d", &w[i]);
    d=gcd(d, w[i]);
  }
  if(d!=1){
    printf("INF");
    return 0;
  }
  memset(dp, 0, sizeof(dp));
  for(int i=0; i<n; i++){
    for(int j=1; j<=10000; j++){
      if(j>=w[i]){
        dp[j] = max(dp[j-w[i]]+w[i], dp[j]);
      }
    }
  }
  int ans=0;
  for(int i=1; i<=10000; i++){
    if(dp[i]!=i) ++ans;
  }
  printf("%d", ans);
  return 0;
}
```
